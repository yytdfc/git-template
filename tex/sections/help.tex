	a_t & = \arg\max\limits_{a}\left(Q(s,a)+u(s,a)\right)\\
	Q(s,a) & =\frac{\bar{L}}{L(s,a)/N(s,a)} \\
	u(s,a) & = C_pP(s,a)\frac{\sqrt{\sum_bN(s,b)}}{1+N(s,a)}
\label{eqn:select}
\end{align}
